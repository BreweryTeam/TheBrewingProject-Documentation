#### Welcome to the TheBrewingProject documentation!

import Aside from "@/components/docs/aside/Aside.astro";

import welcomeImage from "@/assets/img/welcome.png";

export async function Status() {
    const res = await fetch("https://api.mcsrvstat.us/3/showcase.breweryteam.dev");
    const status = await res.json();

    return <div>
            <strong>Status:</strong> {status.online ? <span>
            <span style="color: green;">Online</span> - {status.players.online} / {status.players.max} players, { status.protocol?.name ?? "unknown version" }
            </span> : <span style="color: red;">Offline</span>}
        </div>
}
